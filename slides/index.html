AMD

Direct refs, simpler APIs for your modules

define(['jquery', 'underscore', 'text!./View.html'], function ($, _, text) {
    var template = _.template(text);

    function View(node) {
        this.node = node;
    }

    View.prototype = {
        render: function (data) {
            this.node.innerHTML = template(data);
        }
    };

    return View;
});
